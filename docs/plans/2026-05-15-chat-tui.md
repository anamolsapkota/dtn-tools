**Goal:** Replace the input()-based `dtn chat` with a full-screen urwid TUI featuring split sidebar (Neighbors + Known Nodes), scrollable color-coded message pane, and multiple conversation-switching mechanisms.

```python
        self.known_header_widget = known_header

            "title_bar",
        )

        self.msg_walker = urwid.SimpleFocusListWalker([])
        self.msg_listbox = urwid.ListBox(self.msg_walker)

        # Input bar
        self.input_edit = urwid.Edit(("input_prompt", "❯ "))
        self.input_bar = urwid.AttrMap(self.input_edit, "default")

        # Right pane: chat header + messages + input
        right_pane = urwid.Frame(
            body=self.msg_listbox,
            header=self.chat_header,
            footer=self.input_bar,
        )

        # Columns: sidebar (weight 30) | right pane (weight 70)
        self.columns = urwid.Columns([
            ("weight", 30, self.sidebar),
            ("weight", 70, right_pane),
        ], dividechars=0)

        # Main frame: title bar + columns + status bar
        self.frame = urwid.Frame(
            body=self.columns,
            header=self.title_bar,
            footer=self.status_bar,
        )

    def run(self):
        """Start the TUI event loop."""
        if self.dry_run:
            return

        # Register endpoint
        _run_admin("bpadmin", f"a endpoint {self.recv_eid} q\nq\n")

        self.running = True
        self.loop = urwid.MainLoop(
            self.frame,
            palette=PALETTE,
            unhandled_input=self._handle_input,
            handle_mouse=False,
        )

        # Register pipe for receiver thread wakeup
        self.loop.watch_pipe(self._on_pipe_data)

        # Start receiver
        self.recv_dir = tempfile.mkdtemp(prefix="dtn-chat-")
        self.recv_thread = threading.Thread(
            target=self._receiver_loop, daemon=True
        )
        self.recv_thread.start()

        # Fetch nodes and populate sidebar
        self._refresh_nodes()

        # Resume last conversation
        last = self.history.get_last_active()
        if last:
            self._switch_to(last)

        try:
            self.loop.run()
        finally:
            self.running = False
            if self.active_ipn:
                self.history.set_last_active(self.active_ipn)
            import shutil
            shutil.rmtree(self.recv_dir, ignore_errors=True)
```

```python
# ---------------------------------------------------------------------------
# Sidebar entry widget

class SidebarEntry(urwid.WidgetWrap):
    """A single node entry in the sidebar."""

    def __init__(self, ipn, name, unread=0, extra="", is_active=False):
        self.ipn = ipn
        self.name = name or f"ipn:{ipn}"
        self.unread = unread
        self.extra = extra

        parts = []
        parts.append(("default", f" {self.name}"))
        if unread > 0:
            parts.append(("unread_count", f" ({unread})"))
        if extra:
            parts.append(("msg_ts", f" {extra}"))

        text = urwid.Text(parts)
        if is_active:
            mapped = urwid.AttrMap(text, "active_conv", focus_map="active_conv")
        else:
            mapped = urwid.AttrMap(text, "default", focus_map="active_conv")
        super().__init__(mapped)

    def selectable(self):
        return True

    def keypress(self, size, key):
        return key
```

```python
def _populate_sidebar(self, neighbors, known):
    """Fill sidebar with neighbor and known node entries."""
    self.neighbor_walker.clear()
    for ipn, info in sorted(neighbors.items(), key=lambda x: x[1].get("name", x[0])):
        name = info.get("name", "")
        unread = self.history.unread_count(ipn)
        extra = info.get("outduct", "").split(":")[0] if info.get("outduct") else ""
        is_active = ipn == self.active_ipn
        self.neighbor_walker.append(SidebarEntry(ipn, name, unread, extra, is_active))

    self.known_walker.clear()
    for ipn, info in sorted(known.items(), key=lambda x: x[1].get("name", x[0])):
        name = info.get("name", "")
        unread = self.history.unread_count(ipn)
        hops = info.get("hops", "?")
        extra = f"{hops} hop{'s' if hops != 1 else ''}"
        is_active = ipn == self.active_ipn
        self.known_walker.append(SidebarEntry(ipn, name, unread, extra, is_active))

    # Update headers
    self.neighbor_header_widget.original_widget.set_text(
        f" ● NEIGHBORS ({len(neighbors)})"
    )
    self.known_header_widget.original_widget.set_text(
        f" ◆ KNOWN ({len(known)})"
    )

def _refresh_nodes(self):
    """Fetch contact graph and plans from ION, populate sidebar."""
    if self.dry_run:
        return

    # Get plans (neighbors)
    out, _, _ = _run_admin("ipnadmin", "l plan\nq")
    neighbors = {}
    for line in out.splitlines():
        line = line.strip().lstrip(":").strip()
        m = re.match(r"(\d+)\s+xmit\s+(\S+)", line)
        if m and m.group(1) != self.my_ipn:
            ipn = m.group(1)
            outduct = m.group(2)
            name = self.node_names.get(ipn, "")
            neighbors[ipn] = {"name": name, "outduct": outduct}
            self.plans[ipn] = outduct

    # Get all nodes from contacts
    out, _, _ = _run_admin("ionadmin", "l contact\nq")
    all_ipns = set()
    for line in out.splitlines():
        m_c = re.search(r"node\s+(\d+)\s+to\s+node\s+(\d+)", line)
        if m_c:
            all_ipns.add(m_c.group(1))
            all_ipns.add(m_c.group(2))
    all_ipns.discard(self.my_ipn)

    # Known = all minus neighbors
    known = {}
    for ipn in sorted(all_ipns):
        if ipn not in neighbors:
            name = self.node_names.get(ipn, "")
            known[ipn] = {"name": name, "hops": "?"}

    self.node_list = sorted(all_ipns, key=lambda x: int(x))
    self._populate_sidebar(neighbors, known)

    # Update network status
    total_contacts = sum(1 for line in out.splitlines() if "xmit rate" in line)
    self.net_status_widget.original_widget.set_text(
        f" ● ION running | {len(neighbors)}n · {len(all_ipns)} nodes · {total_contacts} contacts"
    )
```

```python
def _append_message(self, msg):
    """Render a single message dict and append to the message walker."""
    ts_str = msg.get("ts", "")
    try:
        dt = datetime.fromisoformat(ts_str)
    except (ValueError, TypeError):
        dt = datetime.now(timezone.utc)

    # Date separator
    date_key = dt.strftime("%Y-%m-%d")
    if not hasattr(self, "_last_date"):
        self._last_date = None
    if date_key != self._last_date:
        self._last_date = date_key
        date_text = dt.strftime("%B %d, %Y")
        sep = urwid.Text(("date_sep", f"  — {date_text} —"), align="center")
        sep._is_date_sep = True
        self.msg_walker.append(sep)

    # Message line
    ts_display = dt.strftime("%H:%M")
    if msg.get("dir") == "out":
        name_attr = "msg_you"
        name_text = "you"
    else:
        name_attr = "msg_them"
        name_text = msg.get("name") or msg.get("from", "?")

    line = urwid.Text([
        ("msg_ts", f"  {ts_display} "),
        (name_attr, f"{name_text}"),
        ("default", f": {msg.get('msg', '')}"),
    ])
    self.msg_walker.append(line)

def _load_conversation(self, ipn):
    """Load messages from history into the message pane."""
    self.msg_walker.clear()
    self._last_date = None
    messages = self.history.get_recent(ipn, 50)
    for msg in messages:
        self._append_message(msg)
    # Scroll to bottom
    if self.msg_walker:
        self.msg_listbox.set_focus(len(self.msg_walker) - 1)

def _scroll_to_bottom(self):
    """Scroll message pane to the most recent message."""
    if self.msg_walker:
        self.msg_listbox.set_focus(len(self.msg_walker) - 1)
```

```python
def _handle_input(self, key):
    """Handle unhandled keystrokes (keys not consumed by focused widget)."""
    if key in ("f10", "ctrl c"):
        raise urwid.ExitMainLoop()

    if key == "f1":
        self._show_help()
        return

    if key == "f2":
        self._refresh_nodes()
        return

    if key == "tab":
        self.sidebar_focused = not self.sidebar_focused
        if self.sidebar_focused:
            self.columns.set_focus(0)
            self.sidebar.original_widget.set_attr_map({None: "sidebar_focused"})
        else:
            self.columns.set_focus(1)
            self.sidebar.original_widget.set_attr_map({None: "default"})
        self._update_status_bar()
        return

    if key == "esc" and self.sidebar_focused:
        self.sidebar_focused = False
        self.columns.set_focus(1)
        self._update_status_bar()
        return

    if key == "enter" and self.sidebar_focused:
        # Get selected sidebar entry
        if self.columns.focus_position == 0:
            focused = self.sidebar.original_widget.focus
            if hasattr(focused, 'focus'):
                widget = focused.focus
                if hasattr(widget, 'ipn'):
                    self._switch_to(widget.ipn)
                    self.sidebar_focused = False
                    self.columns.set_focus(1)
        return

    if key == "enter" and not self.sidebar_focused:
        text = self.input_edit.get_edit_text().strip()
        if not text:
            return
        self.input_edit.set_edit_text("")
        if text.startswith("/"):
            self._process_command(text)
        else:
            self._send_message(text)
        return

    if key == "meta up":  # Alt+Up
        self._cycle_conversation(-1)
        return

    if key == "meta down":  # Alt+Down
        self._cycle_conversation(1)
        return

    if key == "meta n":  # Alt+N
        self._jump_to_unread()
        return

    if key in ("page up", "page down"):
        # Forward to message listbox
        self.msg_listbox.keypress((20,), key)
        return

def _switch_to(self, ipn):
    """Switch active conversation to ipn."""
    ipn = str(ipn).replace("ipn:", "")
    self.active_ipn = ipn
    self.history.mark_read(ipn)
    self.history.set_last_active(ipn)

    # Update chat header
    name = self.node_names.get(ipn, f"ipn:{ipn}")
    if ipn in self.plans:
        conn_info = f"● direct · {self.plans[ipn]}"
    else:
        conn_info = "◇ via contact graph"
    self.chat_header.original_widget.set_text(
        f" {name}  ipn:{ipn}  {conn_info}"
    )

    # Load messages
    self._load_conversation(ipn)

    # Refresh sidebar to update active highlight
    self._refresh_nodes()
    self._update_status_bar()

def _send_message(self, text):
    """Send a chat bundle to the active conversation."""
    if not self.active_ipn:
        self._set_status("No active conversation. Use /to <node> or Tab to select.")
        return

    dest = f"ipn:{self.active_ipn}.{self.CHAT_SVC}"
    payload = json.dumps({
        "from": self.my_ipn,
        "name": self.my_name,
        "msg": text,
        "ts": time.strftime("%H:%M:%S"),
    })
    escaped = payload.replace("'", "'\\''")
    _, _, rc = _run(f"bpsource {dest} '{escaped}' 2>/dev/null")

    if rc == 0:
        self.history.add_outgoing(self.active_ipn, self.my_ipn, self.my_name, text)
        self._append_message({
            "dir": "out", "from": self.my_ipn, "name": self.my_name,
            "msg": text, "ts": datetime.now(timezone.utc).isoformat(), "read": True,
        })
        self._scroll_to_bottom()
    else:
        self._set_status("Send failed!")

def _process_command(self, text):
    """Handle /commands typed in the input bar."""
    cmd = text.lstrip("/").strip()
    parts = cmd.split(None, 1)
    command = parts[0].lower() if parts else ""

    if command in ("quit", "exit", "q"):
        raise urwid.ExitMainLoop()

    elif command == "to" and len(parts) > 1:
        target = parts[1].strip().replace("ipn:", "")
        # Try name match
        for ipn, name in self.node_names.items():
            if name.lower() == target.lower() or target.lower() in name.lower():
                self._switch_to(ipn)
                return
        # Try as IPN number
        if target.isdigit():
            self._switch_to(target)
            return
        self._set_status(f"Unknown node: {target}")

    elif command == "nodes":
        self._refresh_nodes()
        self._set_status("Node list refreshed")

    elif command == "help":
        self._show_help()

    else:
        self._set_status(f"Unknown command: /{command}. Type /help for commands.")

def _cycle_conversation(self, direction):
    """Cycle through conversations by recency. direction: +1 forward, -1 back."""
    convos = self.history.list_conversations()
    if not convos:
        return
    ipns = [c[0] for c in convos]
    if self.active_ipn in ipns:
        idx = ipns.index(self.active_ipn)
        idx = (idx + direction) % len(ipns)
    else:
        idx = 0
    self._switch_to(ipns[idx])

def _jump_to_unread(self):
    """Jump to the next conversation with unread messages."""
    unread = self.history.all_unread()
    if not unread:
        self._set_status("No unread messages")
        return
    # Pick first unread that's not the active conversation
    for ipn in unread:
        if ipn != self.active_ipn:
            self._switch_to(ipn)
            return
    # All unread are in current conversation
    self._set_status("Already in the only unread conversation")

def _update_status_bar(self):
    """Update the status bar with current conversation info."""
    if self.active_ipn:
        name = self.node_names.get(self.active_ipn, f"ipn:{self.active_ipn}")
        if self.active_ipn in self.plans:
            info = f"{name} · direct"
        else:
            info = f"{name} · routed"
    else:
        info = "No conversation"

    hints = "↑↓:scroll  Tab:sidebar  Alt+↑↓:switch  PgUp/Dn:messages  /:cmd  Ctrl+C:quit"
    self.status_bar.original_widget.set_text(f" {info}    {hints}")

def _set_status(self, text):
    """Set a temporary status message."""
    self.status_bar.original_widget.set_text(f" {text}")

def _show_help(self):
    """Show help overlay."""
    help_text = urwid.Text([
        ("title_bar", " DTN Chat — Help\n\n"),
        ("default", " Keybindings:\n"),
        ("msg_you", "   Tab        "), ("default", "Toggle sidebar focus\n"),
        ("msg_you", "   Enter      "), ("default", "Send message / Select node\n"),
        ("msg_you", "   Alt+↑/↓   "), ("default", "Cycle conversations\n"),
        ("msg_you", "   Alt+N      "), ("default", "Next unread conversation\n"),
        ("msg_you", "   PgUp/Down  "), ("default", "Scroll messages\n"),
        ("msg_you", "   F1         "), ("default", "This help\n"),
        ("msg_you", "   F2         "), ("default", "Refresh node list\n"),
        ("msg_you", "   F10        "), ("default", "Quit\n\n"),
        ("default", " Commands:\n"),
        ("msg_you", "   /to <name> "), ("default", "Switch conversation\n"),
        ("msg_you", "   /nodes     "), ("default", "Refresh nodes\n"),
        ("msg_you", "   /help      "), ("default", "Show this help\n"),
        ("msg_you", "   /quit      "), ("default", "Exit\n\n"),
        ("msg_ts", " Press any key to close"),
    ])
    filler = urwid.Filler(help_text, valign="middle")
    overlay = urwid.Overlay(
        urwid.LineBox(filler, title="Help"),
        self.frame,
        align="center", width=("relative", 50),
        valign="middle", height=("relative", 60),
    )
    self.loop.widget = overlay
    # Any key dismisses
    def dismiss(key):
        self.loop.widget = self.frame
        self.loop.unhandled_input = self._handle_input
    self.loop.unhandled_input = dismiss
```

```python
def _receiver_loop(self):
    """Background: run bprecvfile, poll for files, push to urwid via pipe."""
    proc = subprocess.Popen(
        ["bprecvfile", self.recv_eid],
        cwd=self.recv_dir,
        stdout=subprocess.DEVNULL,
        stderr=subprocess.DEVNULL,
    )
    while self.running:
        time.sleep(0.5)
        if proc.poll() is not None and self.running:
            proc = subprocess.Popen(
                ["bprecvfile", self.recv_eid],
                cwd=self.recv_dir,
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
            )
        try:
            for fname in sorted(os.listdir(self.recv_dir)):
                fpath = os.path.join(self.recv_dir, fname)
                if not os.path.isfile(fpath):
                    continue
                try:
                    with open(fpath) as f:
                        content = f.read().strip()
                    os.unlink(fpath)
                    if not content:
                        continue

                    # Parse message
                    try:
                        data = json.loads(content)
                        sender_ipn = str(data.get("from", "?"))
                        sender_name = data.get("name", "")
                        msg = data.get("msg", content)
                    except json.JSONDecodeError:
                        sender_ipn = "unknown"
                        sender_name = ""
                        msg = content

                    if sender_name and sender_ipn != "unknown":
                        self.node_names[sender_ipn] = sender_name

                    # Queue message and wake urwid
                    msg_data = {
                        "dir": "in",
                        "from": sender_ipn,
                        "name": sender_name,
                        "msg": msg,
                        "ts": datetime.now(timezone.utc).isoformat(),
                        "read": sender_ipn == self.active_ipn,
                    }
                    with self._pending_lock:
                        self._pending_messages.append((sender_ipn, sender_name, msg_data))
                    os.write(self._pipe_w, b"1")

                except Exception:
                    pass
        except Exception:
            pass
    proc.terminate()

def _on_pipe_data(self, data):
    """Called by urwid main loop when receiver thread signals new data."""
    with self._pending_lock:
        messages = list(self._pending_messages)
        self._pending_messages.clear()

    for sender_ipn, sender_name, msg_data in messages:
        if sender_ipn == self.active_ipn:
            # Active conversation — show inline, mark read
            self.history.add_incoming(sender_ipn, sender_name, msg_data["msg"], read=True)
            self._append_message(msg_data)
            self._scroll_to_bottom()
        else:
            # Other conversation — save as unread, update sidebar
            self.history.add_incoming(sender_ipn, sender_name, msg_data["msg"], read=False)
            unread = self.history.unread_count(sender_ipn)
            label = sender_name or f"ipn:{sender_ipn}"
            self._set_status(f"← {label} sent a message ({unread} unread)")

        # Refresh sidebar to update unread counts
        self._refresh_nodes()

    return True  # Keep watching pipe
```
